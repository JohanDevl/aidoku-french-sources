<script>
    /**
     * Anime Welcome Popup Handler
     * 
     * This script handles the anime-themed welcome popup that appears
     * every 7 days for users. It includes animations and cookie management.
     */
    
    // Custom animations for the popup using Web Animations API
    const animePopupAnimations = {
        // Bounce in animation
        bounceIn: [
            { transform: 'scale(0)', opacity: 0, offset: 0 },
            { transform: 'scale(1.05)', opacity: 1, offset: 0.8 },
            { transform: 'scale(0.95)', opacity: 1, offset: 0.9 },
            { transform: 'scale(1)', opacity: 1, offset: 1 }
        ],
        
        // Fade out animation
        fadeOut: [
            { transform: 'scale(1)', opacity: 1, offset: 0 },
            { transform: 'scale(0.95)', opacity: 0.9, offset: 0.2 },
            { transform: 'scale(0)', opacity: 0, offset: 1 }
        ]
    };
    
    // Show popup with animation when page loads (if no cookie exists)
    document.addEventListener('DOMContentLoaded', function() {
        const popup = document.getElementById('anime-welcome-popup');
        const content = document.getElementById('popup-content');
        if (!getAnimeCookie('anime_welcome_popup_seen')) {
           
            
            if (!popup || !content) return;
            
            // Display the container
            popup.style.display = 'flex';
            
            // Add slight delay before animation for better effect
            setTimeout(function() {
                // Use Web Animations API for more control if supported
				
                if (typeof content.animate === 'function') {
                    content.animate(animePopupAnimations.bounceIn, {
                        duration: 900,
                        easing: 'cubic-bezier(0.25, 0.46, 0.45, 0.94)',
                        fill: 'forwards'
                    });
                }
                
                // Also add classes for browsers that might not support Web Animations API
                content.classList.remove('scale-0');
                content.classList.add('scale-100');
                content.classList.add('animate-bounce-in');
            }, 900);
        }
        else{
  
            popup.style.display = 'none';
        }
		
		
    });
    
    // Close popup with animation and set cookie
    function closeAnimePopup() {
        const popup = document.getElementById('anime-welcome-popup');
        const content = document.getElementById('popup-content');
        
        if (!popup || !content) return;
        
        // Use Web Animations API for the exit animation if supported
        if (typeof content.animate === 'function') {
            const animation = content.animate(animePopupAnimations.fadeOut, {
                duration: 500,
                easing: 'ease-out',
                fill: 'forwards'
            });
            
            // Wait for animation to complete before hiding
            animation.onfinish = function() {
                popup.style.display = 'none';
            };
        } else {
            // Fallback for browsers without Web Animations API
            content.classList.remove('scale-100');
            content.classList.remove('animate-bounce-in');
            content.classList.add('scale-0');
            content.classList.add('animate-fade-out');
            
            // Wait for animation to complete before hiding
            setTimeout(function() {
                popup.style.display = 'none';
            }, 500);
        }
        
        // Set cookie for 7 days
        setAnimeCookie('anime_welcome_popup_seen', '1', 7);
    }
    
    // Helper function to set a cookie
    function setAnimeCookie(name, value, days) {
        var expires = "";
        if (days) {
            var date = new Date();
            date.setTime(date.getTime() + (days * 24 * 60 * 60 * 1000));
            expires = "; expires=" + date.toUTCString();
        }
        document.cookie = name + "=" + value + expires + "; path=/; SameSite=Lax";
    }
    
    // Helper function to get a cookie
    function getAnimeCookie(name) {
        var nameEQ = name + "=";
        var ca = document.cookie.split(';');
        for(var i=0; i < ca.length; i++) {
            var c = ca[i];
            while (c.charAt(0) == ' ') c = c.substring(1, c.length);
            if (c.indexOf(nameEQ) == 0) return c.substring(nameEQ.length, c.length);
        }
        return null;
    }
